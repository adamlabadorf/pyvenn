[docs]def quadratic_bezier_point(p0,p1,p2,t) :
    return (1-t)**2*p0 + 2*(1-t)*t*p1 + t**2*p2



[docs]def quadratic_bezier_curve(p0,p1,p2,n=10) :
    """p0, p1, and p2 are length 2 vectors, n is the number of points making up the curve"""
    curve_pts = []
    for t in arange(0,1+1./n,1./n) :
        curve_pts.append(quadratic_bezier_point(p0,p1,p2,t))
    return curve_pts



[docs]def cubic_bezier_curve(p0,p1,p2,p3,n) :
    """p0, p1, p2, and p3 are length 2 vectors, n is the number of points making up the curve"""
    curve_pts = []
    for t in arange(0,1+1./n,1./n) :
        q_point_1 = quadratic_bezier_point(p0,p1,p2,t)
        q_point_2 = quadratic_bezier_point(p1,p2,p3,t)
        curve_pts.append((1-t)*q_point_1 + t*q_point_2)
    return curve_pts



[docs]def bezier_circle(origin,radius,n) :
    
    orig_pts = array(((0,1),(1,1),(1,0)))

    rot_mat_f = lambda theta : array(((cos(theta),-sin(theta)),(sin(theta),cos(theta))))
    rotations = [0,pi/2,pi,3*pi/2]
    circle_pts = []
    for rot in rotations :
        rot_mat = rot_mat_f(rot)
        rot_pts = dot(orig_pts,rot_mat)
        circle_pts.extend(quadratic_bezier_curve(rot_pts[0],rot_pts[1],rot_pts[2],n/4.))
    circle_pts = [x+origin for x in circle_pts]
    return circle_pts



[docs]def trig_circle(origin,radius,n) :
    circle_pts = []
    for t in arange(0,2*pi+1./n,1./n) :
        x = origin[0] + radius*cos(t)
        y = origin[1] + radius*sin(t)
        circle_pts.append((x,y))
    return circle_pts



[docs]def find_best_d(A,B,C) :
    """A - total # in A, B - total # in B, C - total # in overlap"""

    assert C <= A and C <= B

    # the larger class gets a area of 1,
    # the smaller class gets a area of <= 1
    # intersection gets area <= area smaller class
    max_class = float(max(A,B))
    A_area = A/max_class
    B_area = B/max_class
    C_area = C/max_class
    print A, B, C
    print A_area, B_area, C_area

    A_radius = sqrt(A_area/pi)
    B_radius = sqrt(B_area/pi)

    best_d, best_area_diff = 0., 1e5
    best_vars = []
    best_intersect_points = []
    for d in arange(0,A_radius+B_radius,0.01) :
        d_m = (A_radius+B_radius-d)
        m_1 = (d**2 - B_radius**2 + A_radius**2)/(2*d)
        try :
            h = sqrt(A_radius**2-m_1**2)
        except :
            # complex number, not right
            continue
        m_2 = d-m_1

        # calculate lens area
        c_1_area = A_radius**2*arccos(m_1/A_radius)-m_1*sqrt(A_radius**2-m_1**2)
        c_2_area = B_radius**2*arccos(m_2/B_radius)-m_2*sqrt(B_radius**2-m_2**2)

        computed_area = c_1_area + c_2_area

        area_diff = abs(C_area-computed_area)
        if area_diff < best_area_diff :
            best_d = d
            best_area_diff = area_diff
            best_intersect_points = ((m_1,h),(m_1,-h))

    return A_radius, B_radius, best_d, best_intersect_points
